```python
@router.post("/api/upload")
async def upload_pdf(file: UploadFile = File(...)):
    if not file.filename or not file.filename.lower().endswith(".pdf"):
        raise HTTPException(status_code=400, detail="Only PDF files are accepted")
    PAPERS_ROOT.mkdir(parents=True, exist_ok=True)
    stem = Path(file.filename).stem
    filename = file.filename
    target = PAPERS_ROOT / filename
    if target.exists():
        filename = f"{stem}_{int(time.time())}.pdf"
        target = PAPERS_ROOT / filename
    with open(target, "wb") as fh:
        shutil.copyfileobj(file.file, fh)
    return {"path": f"papers/{filename}", "filename": filename}


@router.get("/api/papers/{filename}")
async def get_paper(filename: str):
    PAPERS_ROOT.mkdir(parents=True, exist_ok=True)
    file_path = (PAPERS_ROOT / filename).resolve()
    try:
        file_path.relative_to(PAPERS_ROOT.resolve())
    except ValueError:
        raise HTTPException(status_code=403, detail="Access denied")
    if not file_path.is_file():
        raise HTTPException(status_code=404, detail=f"File {filename} not found")
    return FileResponse(file_path)
```

```typescript
const BASE = ''  // same origin in dev (Vite proxies /api → :8000)

export interface Task {
  task_id: string
  status: 'pending' | 'running' | 'done' | 'error' | 'interrupted'
  created_at: string
  pdf_path?: string | null
}

export interface TaskDetail extends Task {
  artifacts: string[]
}

export async function listTasks(): Promise<Task[]> {
  const res = await fetch(`${BASE}/api/tasks`)
  if (!res.ok) throw new Error(await res.text())
  return res.json()
}

export async function getTask(taskId: string): Promise<TaskDetail> {
  const res = await fetch(`${BASE}/api/tasks/${taskId}`)
  if (!res.ok) throw new Error(await res.text())
  return res.json()
}

export async function uploadPdf(file: File): Promise<{ path: string; filename: string }> {
  const form = new FormData()
  form.append('file', file)
  const res = await fetch(`${BASE}/api/upload`, { method: 'POST', body: form })
  if (!res.ok) throw new Error(await res.text())
  return res.json()
}

export async function createTask(params: {
  pdf_path: string
  fast: boolean
  no_critique: boolean
}): Promise<Task> {
  const res = await fetch(`${BASE}/api/tasks`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(params),
  })
  if (!res.ok) throw new Error(await res.text())
  return res.json()
}

export async function getArtifactText(taskId: string, artifactPath: string): Promise<string> {
  const res = await fetch(`${BASE}/api/tasks/${taskId}/artifacts/${artifactPath}`)
  if (!res.ok) throw new Error(await res.text())
  return res.text()
}
```

```typescript
import { useState, useEffect, useCallback } from 'react'
import { listTasks, Task } from '../api/client'

export function useTasks(pollIntervalMs = 2000) {
  const [tasks, setTasks] = useState<Task[]>([])
  const [error, setError] = useState<string | null>(null)

  const refresh = useCallback(async () => {
    try {
      const data = await listTasks()
      setTasks([...data].reverse())
      setError(null)
    } catch (e) {
      setError(String(e))
    }
  }, [])

  useEffect(() => {
    refresh()
    const iv = setInterval(refresh, pollIntervalMs)
    return () => clearInterval(iv)
  }, [refresh, pollIntervalMs])

  return { tasks, error, refresh }
}
```

- [ ] **Step 3: Write `frontend/src/hooks/useSSE.ts`**

```typescript
import { useState, useEffect } from 'react'

export interface SSEEvent {
  type: 'progress' | 'file_written' | 'done' | 'error'
  pct?: number
  message?: string
  phase?: string
  path?: string
  section_ref?: string
  summary?: string
  ts?: string
}

export function useSSE(taskId: string | null) {
  const [events, setEvents] = useState<SSEEvent[]>([])
  const [connected, setConnected] = useState(false)
  const [done, setDone] = useState(false)

  useEffect(() => {
    if (!taskId) return
    setEvents([])
    setDone(false)

    const es = new EventSource(`/api/tasks/${taskId}/events`)
    setConnected(true)

    es.onmessage = (e) => {
      try {
        const evt = JSON.parse(e.data) as SSEEvent
        setEvents(prev => [...prev, evt])
        if (evt.type === 'done' || evt.type === 'error') {
          setDone(true)
          es.close()
          setConnected(false)
        }
      } catch {}
    }

    es.onerror = () => {
      setConnected(false)
    }

    return () => {
      es.close()
      setConnected(false)
    }
  }, [taskId])

  return { events, connected, done }
}
```

```tsx
import { useState, useRef, DragEvent } from 'react'
import { useNavigate } from 'react-router-dom'
import { uploadPdf, createTask } from '../api/client'

interface Props {
  onTaskCreated: () => void
}

export default function UploadPanel({ onTaskCreated }: Props) {
  const nav = useNavigate()
  const inputRef = useRef<HTMLInputElement>(null)
  const [dragging, setDragging] = useState(false)
  const [file, setFile] = useState<File | null>(null)
  const [fast, setFast] = useState(false)
  const [noCritique, setNoCritique] = useState(false)
  const [loading, setLoading] = useState(false)
  const [error, setError] = useState<string | null>(null)

  function handleFile(f: File) {
    if (!f.name.toLowerCase().endsWith('.pdf')) {
      setError('请选择 PDF 文件')
      return
    }
    if (f.size > 200 * 1024 * 1024) {
      setError('文件不能超过 200MB')
      return
    }
    setError(null)
    setFile(f)
  }

  function onDrop(e: DragEvent) {
    e.preventDefault()
    setDragging(false)
    const f = e.dataTransfer.files[0]
    if (f) handleFile(f)
  }

  async function submit() {
    if (!file) return
    setLoading(true)
    setError(null)
    try {
      const { path } = await uploadPdf(file)
      const task = await createTask({ pdf_path: path, fast, no_critique: noCritique })
      onTaskCreated()
      nav(`/app?task=${task.task_id}`)
    } catch (e) {
      setError(String(e))
    } finally {
      setLoading(false)
    }
  }

                 onChange={e => e.target.files?.[0] && handleFile(e.target.files[0])} />
```

```tsx
import { useState, useEffect } from 'react'
import { getArtifactText, TaskDetail } from '../../api/client'

interface Props {
  detail: TaskDetail
}

interface CorpusStats {
  llm_turns: number
  tool_calls: number
  total_tokens: number
}

async function parseCorpusStats(detail: TaskDetail): Promise<CorpusStats> {
  const llmPath = detail.artifacts.find(a => a.includes('logs/llm.jsonl'))
  if (!llmPath) return { llm_turns: 0, tool_calls: 0, total_tokens: 0 }
  const text = await getArtifactText(detail.task_id, llmPath)
  const lines = text.trim().split('\n').filter(Boolean)
  let toolCalls = 0
  let totalTokens = 0
  for (const line of lines) {
    try {
      const obj = JSON.parse(line)
      if (obj.role === 'assistant' && Array.isArray(obj.content)) {
        toolCalls += obj.content.filter((c: { type: string }) => c.type === 'tool_use').length
      }
      if (obj.usage) {
        totalTokens += (obj.usage.input_tokens ?? 0) + (obj.usage.output_tokens ?? 0)
      }
    } catch {}
  }
  return { llm_turns: lines.length, tool_calls: toolCalls, total_tokens: totalTokens }
}

export default function CorpusTab({ detail }: Props) {
  const [stats, setStats] = useState<CorpusStats | null>(null)

  useEffect(() => {
    parseCorpusStats(detail).then(setStats).catch(() => setStats({ llm_turns: 0, tool_calls: 0, total_tokens: 0 }))
  }, [detail.task_id])

  const cards = stats ? [
    { label: 'LLM 调用', value: stats.llm_turns, color: 'var(--blue)' },
    { label: '工具调用', value: stats.tool_calls, color: 'var(--green)' },
    { label: '累计 Token', value: stats.total_tokens > 1000 ? `${(stats.total_tokens / 1000).toFixed(1)}K` : String(stats.total_tokens), color: 'var(--navy)' },
    { label: '代码文件', value: detail.artifacts.filter(a => a.endsWith('.py')).length, color: 'var(--amber)' },
  ] : []

  return (
    <div className="p-6">
      <h3 className="text-sm font-bold mb-4" style={{ color: 'var(--navy)' }}>语料统计</h3>
      {!stats ? (
        <p className="text-sm" style={{ color: 'var(--muted)' }}>加载中…</p>
      ) : (
        <div className="grid grid-cols-2 gap-4 max-w-lg">
          {cards.map(c => (
            <div key={c.label} className="rounded-xl p-5 shadow-sm"
                 style={{ background: 'var(--surface)', border: '1px solid var(--border-lt)' }}>
              <div className="text-xs font-bold uppercase tracking-wide mb-2" style={{ color: 'var(--muted)' }}>
                {c.label}
              </div>
              <div className="text-3xl font-extrabold" style={{ color: c.color }}>{c.value}</div>
            </div>
          ))}
        </div>
      )}
      <p className="mt-6 text-xs" style={{ color: 'var(--muted)' }}>
        完整交互数据保存于 <span className="mono">output/tasks/{detail.task_id}/logs/llm.jsonl</span>
      </p>
    </div>
  )
}
```

```tsx
import { useEffect, useRef, useState } from 'react'
import * as pdfjsLib from 'pdfjs-dist'
import type { PDFDocumentProxy, PDFPageProxy } from 'pdfjs-dist'

interface Props {
  pdfUrl: string           // e.g. "/api/papers/lora.pdf"
  searchText?: string      // text to highlight
  onClose: () => void
}

export default function PdfViewer({ pdfUrl, searchText, onClose }: Props) {
  const canvasRef = useRef<HTMLCanvasElement>(null)
  const [pdf, setPdf] = useState<PDFDocumentProxy | null>(null)
  const [pageNum, setPageNum] = useState(1)
  const [numPages, setNumPages] = useState(0)
  const [loading, setLoading] = useState(true)
  const [error, setError] = useState<string | null>(null)

  // Load PDF
  useEffect(() => {
    setLoading(true)
    setError(null)
    pdfjsLib.getDocument(pdfUrl).promise
      .then(doc => {
        setPdf(doc)
        setNumPages(doc.numPages)
        setLoading(false)
      })
      .catch(() => {
        setError('无法加载 PDF')
        setLoading(false)
      })
    return () => { setPdf(null) }
  }, [pdfUrl])

  // Find page containing searchText
  useEffect(() => {
    if (!pdf || !searchText) return
    async function findPage() {
      if (!pdf) return
      for (let i = 1; i <= pdf.numPages; i++) {
        const page: PDFPageProxy = await pdf.getPage(i)
        const tc = await page.getTextContent()
        const pageText = tc.items.map((item: { str?: string }) => item.str ?? '').join(' ')
        if (pageText.includes(searchText.slice(0, 30))) {
          setPageNum(i)
          return
        }
      }
    }
    findPage()
  }, [pdf, searchText])

  // Render page
  useEffect(() => {
    if (!pdf || !canvasRef.current) return
    pdf.getPage(pageNum).then((page: PDFPageProxy) => {
      const viewport = page.getViewport({ scale: 1.4 })
      const canvas = canvasRef.current!
      canvas.width = viewport.width
      canvas.height = viewport.height
      page.render({ canvasContext: canvas.getContext('2d')!, viewport })
    })
  }, [pdf, pageNum])

  return (
    <div className="flex flex-col h-full" style={{ borderLeft: '1px solid var(--border-lt)', background: 'var(--surface)' }}>
      {/* Header */}
      <div className="shrink-0 flex items-center justify-between px-3 py-2"
           style={{ borderBottom: '1px solid var(--border-lt)', background: '#1e293b' }}>
        <span className="text-xs mono" style={{ color: '#94a3b8' }}>
          {pdfUrl.split('/').pop()} {numPages > 0 && `· p.${pageNum}/${numPages}`}
        </span>
        <button onClick={onClose} className="text-xs" style={{ color: '#60a5fa' }}>× 关闭</button>
      </div>

      {/* Content */}
      <div className="flex-1 overflow-auto p-2">
        {loading && <p className="text-xs p-4" style={{ color: 'var(--muted)' }}>加载中…</p>}
        {error && <p className="text-xs p-4" style={{ color: 'var(--red)' }}>{error}</p>}
        {!loading && !error && <canvas ref={canvasRef} className="w-full" />}
      </div>

      {/* Pagination */}
      {numPages > 1 && (
        <div className="shrink-0 flex items-center justify-center gap-3 py-2"
             style={{ borderTop: '1px solid var(--border-lt)' }}>
          <button onClick={() => setPageNum(p => Math.max(1, p - 1))} disabled={pageNum === 1}
                  className="text-xs px-2 py-1 rounded disabled:opacity-30"
                  style={{ background: 'var(--border-lt)', color: 'var(--slate)' }}>← 上页</button>
          <span className="text-xs" style={{ color: 'var(--muted)' }}>{pageNum} / {numPages}</span>
          <button onClick={() => setPageNum(p => Math.min(numPages, p + 1))} disabled={pageNum === numPages}
                  className="text-xs px-2 py-1 rounded disabled:opacity-30"
                  style={{ background: 'var(--border-lt)', color: 'var(--slate)' }}>下页 →</button>
        </div>
      )}
    </div>
  )
}
```

```tsx
import { useState, useEffect, useRef } from 'react'
import { getArtifactText, TaskDetail } from '../../api/client'
import PdfViewer from '../PdfViewer'

interface CorrespondenceRow {
  section_ref: string
  original_text: string
  highlight?: string
  critique_type: 'trap' | 'assumption' | 'question'
  critique_text: string
  code_snippet: string
}

const TYPE_LABEL = { trap: '陷阱', assumption: '假设', question: '质疑' }
const TYPE_COLOR = { trap: 'var(--red)', assumption: 'var(--blue)', question: 'var(--muted)' }

interface Props {
  detail: TaskDetail
}

export default function CorrespondenceTab({ detail }: Props) {
  const [rows, setRows] = useState<CorrespondenceRow[]>([])
  const [unavailable, setUnavailable] = useState(false)
  const [pdfSearch, setPdfSearch] = useState<{ url: string; text: string } | null>(null)

    getArtifactText(detail.task_id, corrPath)
      .then(text => {

        else setRows(parsed)
      })
      .catch(() => setUnavailable(true))

  if (unavailable) {
    return (

    <div className="flex h-full overflow-hidden">
      {/* 3-column table */}

        </div>

          <div key={i} className="grid grid-cols-3"
               style={{ borderBottom: '1px solid var(--border-lt)' }}>
            {/* Original text */}
            <div className="p-4" style={{ background: '#fafcff' }}>
              <div className="flex items-center gap-2 mb-2">
                <button
                  onClick={() => pdfUrl && setPdfSearch({ url: pdfUrl, text: row.highlight ?? row.original_text.slice(0, 40) })}

                      <span style={{ color: 'var(--muted)' }}>{row.original_text.split(row.highlight)[1]}</span>
                    </>
                  : <span style={{ color: 'var(--muted)' }}>{row.original_text}</span>

              </pre>
            </div>
          </div>

      </div>

      {/* PDF viewer drawer */}
```
